\begin{frame}{Exercice 2}
On sait que l'on a $S_n \varphi \to \psi$, $\mu$-presque partout pour une certaine fonction $\psi \in L^1(\mu)$. On note $G$ l'ensemble des points de $X$ telle que $\lim_n S_n \varphi(x)$ existe. 
\pau \vspace{0.2cm}

Si $x,x' \in X$ on a, puisque $\varphi$ est uniformŽment continue,
$$
|S_n \varphi(x) - S_n\varphi(x')| \leqslant \frac{1}{n} \sum_{k=1}^n |\varphi(f^k(x))-\varphi(f^k(x'))| \leqslant C \varepsilon\left(\dd(x,x')\right) $$
o $\varepsilon(t) \to 0$ quand $t \to 0$. \pau Comme $G$ est dense dans $X$ (car $\mu(U) > 0$ pour tout ouvert non vide $U$), on peut dŽfinir $\psi$ partout par $\psi(x) = \lim_{k} \psi(x_k)$ o $x_k \in G$ et $x_k \to x.$ \pau \vspace{0.2cm}

Ainsi $S_n \varphi \to \psi$ partout, avec $\psi$ continue. \pau \vspace{0.2cm}

Montrons que la convergence est uniforme. Soit $\varepsilon > 0$ et $x_1, \dots, x_N \in X$ tels que $\inf_{i=1, \dots, N} \dd(x,x_i) < \delta$ pour tout $x \in X,$\pau o $\delta > 0$ est choisi de sorte que
$$
|\varphi(x)-\varphi(x')|Ê\leqslant \varepsilon, \quad \dd(x,x') < \delta, \quad x,x'\in X.
$$
\end{frame}


\begin{frame}
Soit $n_0$ assez grand tel que $|S_n\varphi(x_i) - \psi(x_i)| < \varepsilon$ pour tout $n > n_0$ et tout $i = 1,\dots,N.$ \pau \vspace{0.2cm}

\end{aligned}

\end{frame}



\begin{frame}{Exercice 3}
Soit $(\varphi_j)$ une suite de $C^0(X)$ qui est dense. \pau Pour tout $j$, il existe $G_j \subset X$ de mesure totale telle que la limite $\lim_n S_n \varphi_j(x)$ existe pour tout $x\in G_j$. \pau  \vspace{0.2cm}

On dŽfinit $G$ l'ensemble de mesure totale par
$$
G = \bigcap_j G_j.
$$
\pau
Soit maintenant $\varphi \in C^0(X)$, et $x \in G.$ Soit $\varepsilon > 0.$ On prend $j$ tel que $\|\varphi_j - \varphi\|_\infty < \varepsilon.$ \pau Alors pour tous $m,n$,
$$
\begin{aligned}
|S_n \varphi(x) - S_m \varphi(x)| &\leqslant |S_nÊ\varphi(x) - S_n \varphi_j(x)|Ê+ |S_n\varphi_j(x) - S_m \varphi_j(x)|Ê\\
& \hspace{4.6cm} \quad + |S_m \varphi_j(x) - S_m\varphi(x)| \pau \\
& \leqslant 2\varepsilon  + |S_n\varphi_j(x) - S_m\varphi_j(x)|.
\end{aligned}
$$
\pau
Si $m,n$ sont assez grands alors $|S_n\varphi_j(x) - S_m\varphi_j(x)| \leqslant \varepsilon$ puisque $(S_n\varphi_j(x))_n$ converge car $x \in G \subset G_j.$ \pau \vspace{0.2cm}

Ainsi, $S_n \varphi(x)$ converge quand $n \to +\infty$ par le critre de Cauchy.
\end{frame}

\begin{frame}{Exercice 5}
\textbf{1.} Pour tout $n$ on a 
$$
\int_X |S_n \varphi|^2 \dd \mu = \int_X |\varphi|^2 \dd \mu.
$$
\pau
Par consŽquent on a $\int_X |\bar \varphi|^2 \dd \mu \leqslant \int_X |\varphi|^2 \dd \mu$ par Fatou, et donc $\bar \varphi \in L^2(\mu).$\pau \vspace{0.2cm}

\textbf{2.} Si $|\varphi| \in L^\infty(\mu)$ on a $\int_X |S_n \varphi - \bar \varphi|^2 \dd \mu \to 0$ par le thŽorme de convergence dominŽe.

\pau \vspace{0.2cm} Posons $\varphi_k = \varphi 1_{\{|\varphi|\leqslant k\}}.$ Alors 
$$
\int_X |\varphi|^2 \dd \mu \geqslant \int_X |\varphi - \varphi_k|^2 \dd \mu \geqslant k^2 \mu(\{|\varphi| > k),
$$
\pau
de sorte que 
$$
\mu(\{|\varphi|> k\}) \leqslant \frac{\|\varphi\|_{L^2(\mu)}^2}{k^2}, \quad k > 0.
$$
\pau Il suit que $\varphi_k \to \varphi$ $\mu$ presque partout et donc $\varphi_k \to \varphi$ dans $L^2(\mu)$ par convergence dominŽe.
\end{frame}

\begin{frame}
Soit $\varepsilon > 0$ et $k$ assez grand de sorte que $\|\varphi-\varphi_k\|_{L^2(\mu)} < \varepsilon.$ \pau On a 
$$
\|S_n\varphi - S_m\varphi\|_{2} \leqslant\|S_n\varphi - S_n\varphi_k\|_2 + \|S_n \varphi_k - S_m \varphi_k\|_2 + \|S_m \varphi_k - S_m \varphi\|_2.
$$
On a pour tout $\ell$
$$
\|S_\ell \varphi - S_\ell \varphi_k\|_2 \leqslant \frac{1}{\ell} \sum_{j=1}^\ell \|(\varphi - \varphi_k) \circ f^j\|_2 \leqslant \|\varphi - \varphi_k\| < \varepsilon.
$$
\pau
D'autre part, comme $\varphi_k$ est bornŽe on sait que $S_n \varphi_k$ converge dans $L^2(\mu)$ ;\pau on obtient que si $m,n$ sont assez grands, 
$$
\|S_n \varphi - S_m \varphi\|_2 < 3 \varepsilon.
$$
\pau
Ainsi $(S_n \varphi)$ converge dans $L^2(\mu)$, vers $\bar \varphi.$
\end{frame}
